## 基于openapi3.0的yaml文件生成restful风格的javaapi

maven的pom文件中导入插件openapi-generator-maven-plugin,如下：

```
    <build>
        <plugins>
            <plugin>
                <groupId>org.springframework.boot</groupId>
                <artifactId>spring-boot-maven-plugin</artifactId>
            </plugin>
            <plugin>
                <!-- Plugin that provides API-first development using openapi-generator
                    to generate Spring-MVC endpoint stubs at compile time from an OpenAPI definition
                    file -->
                <groupId>org.openapitools</groupId>
                <artifactId>openapi-generator-maven-plugin</artifactId>
                <version>3.3.1</version>
                <executions>
                    <execution>
                        <goals>
                            <goal>generate</goal>
                        </goals>
                        <configuration>
                            <inputSpec>${project.basedir}/src/swagger/你的yaml名.yaml</inputSpec>
                            <generatorName>spring</generatorName>
                            <apiPackage>com.example.openapigenerator.rest</apiPackage>
                            <modelPackage>com.example.openapigenerator.rest.dto</modelPackage>
                            <output>${project.basedir}</output>
                            <supportingFilesToGenerate>ApiUtil.java</supportingFilesToGenerate>
                            <configOptions>
                                <delegatePattern>true</delegatePattern>
                            </configOptions>
                        </configuration>
                    </execution>
                </executions>
            </plugin>
        </plugins>
    </build>

``` 
将yaml文件放置在inputSpec指定的位置；生成的文件出现在output+apiPackage指定的位置；生成的model文件出现在output+modelPackage指定的位置 

这里不赘述yaml的语法格式，推荐在![SwaggerEditor](editor.swagger.io)中编写yaml文件，可以实时查看api。直接上样例
__yaml文件__

```
    openapi: 3.0.0
    info:
      title: 任务管理API文档
      description: 模块及接口设计文档
      version: '0.1'
      termsOfService: 'http://swagger.io/terms/'
      license:
        name: Apache 2.0
        url: 'http://www.apache.org/licenses/LICENSE-2.0.html'
    servers:
      - url: '127.0.0.1:8080/'
        description: Development server
    
    tags:
     - name: Task
    
    paths:
      /task:
        post:
          tags:
          - Task
          summary: 创建任务
          operationId: createTask
          requestBody:
            required: true
            content:
              application/json:
                schema:
                  $ref: '#/components/schemas/CreateTaskRequest'
          responses:
            '200':
              description: 操作成功
              content:
                application/json:
                  schema:
                    $ref: '#/components/schemas/TaskDto'
            '404':
              description: 服务未找到
              content:
                application/json:
                  schema:
                    type: string
                    default: 服务未找到
        get:
          tags:
          - TaskManage
          summary: 获取任务列表
          operationId: getTaskList
          parameters: 
            - in: path
              name: taskListStatus
              required: true
              description: 任务状态[0.全部任务列表，1.下架任务列表，2.上架任务列表]
              schema:
                type: integer
                format: int32
          responses:
            '200':
              description: 操作成功
              content:
                application/json:
                  schema:
                    $ref: '#/components/schemas/TaskList'
            '404':
              description: 服务未找到
              content:
                application/json:
                  schema:
                    type: string
                    default: 服务未找到
      /task/{id}: 
        get:
          tags:
          - TaskManage
          summary: 获取指定任务详情
          operationId: getTask
          parameters:
            - in: path
              name: id
              required: true
              description: 任务主键
              schema:
                type: integer
                format: int64
          responses:
            '200':
              description: 操作成功
              content:
                application/json:
                  schema:
                    $ref: '#/components/schemas/TaskDto'
            '404':
              description: 服务未找到
              content:
                application/json:
                  schema:
                    type: string
                    default: 服务未找到
                    
    components:
      schemas:
        TaskDto:
          description: 任务详情
          type: object
          properties:
            id:
              type: integer
              format: int64
              description: 任务主键
            taskName:
              description: 任务名称
              type: string
            taskDesc:
              description: 任务描述
              type: string
            status:
              type: integer
              format: int32
              description: 任务状态（1.已下架，2.上架）
            taskCycle:
              description: 任务时效[日常，周常，一次性]
              enum: [DAY,WEEK,ONCE]
            tag:
              description: 任务标签
              type: string
            taskDetail:
              description: 任务详情
              type: string
        TaskBriefDto:
          description: 简略任务信息
          type: object
          properties:
            id:
              type: integer
              format: int64
              description: 任务主键
            taskName:
              description: 任务名称
              type: string
            status:
              type: integer
              format: int32
              description: 任务状态（1.已下架，2.上架）
            taskCycle:
              description: 任务时效[日常，周常，一次性]
              enum: [DAY,WEEK,ONCE]
        CreateTaskRequest:
          description: 创建任务详情对象
          type: object
          properties:
            taskName:
              description: 任务名称
              type: string
            taskDesc:
              description: 任务描述
              type: string
            tag:
              description: 任务标签
              type: string
            taskCycle:
              description: 任务时效[每日，每周，一次性]
              enum: [DAY,WEEK,ONCE]
            taskDetail:
              description: 任务详情
              type: string       
        TaskList:
          description: 获取任务列表
          type: object
          properties:
            total:
              description: 任务总条数
              type: integer
              format: int64
            taskList:
              type: array
              items:
                $ref: '#/components/schemas/TaskBriefDto'
```
然后maven命令打包一下：**mvn package**
